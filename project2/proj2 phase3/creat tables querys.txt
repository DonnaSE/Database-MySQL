1. COMPANY_EMPLOYEE

CREATE TABLE `COMPANY_EMPLOYEE`( 
`employeeId` VARCHAR(10) NOT NULL, 
`FName` VARCHAR(15) NOT NULL, 
`LName` VARCHAR(15) NOT NULL, 
`job_typ` VARCHAR(20) , 
`address` VARCHAR(50), 
PRIMARY KEY(employeeId) )


2. DONOR
DROP TABLE IF EXISTS `DONOR`;
CREATE TABLE `DONOR`(
`email` VARCHAR(30) NOT NULL,
`company_name` VARCHAR(40)NOT NULL,
`phone_number` VARCHAR(20),
`location` VARCHAR(50),
`employeeId` VARCHAR(10),
PRIMARY KEY(email),
FOREIGN KEY(employeeId) REFERENCES COMPANY_EMPLOYEE(employeeId)
);



3. CHARITY_CENTER
DROP TABLE IF EXISTS `CHARITY_CENTER`;
CREATE TABLE `CHARITY_CENTER`(
`email` VARCHAR(30) NOT NULL,
`name` VARCHAR(20)NOT NULL,
`phone_number` VARCHAR(20),
`location` VARCHAR(50),
`employeeId` VARCHAR(10),
PRIMARY KEY(email),
FOREIGN KEY(employeeId) REFERENCES COMPANY_EMPLOYEE(employeeId)
);


4. NEEDY
CREATE TABLE `NEEDY`( 
`Id_number` VARCHAR(10) NOT NULL, 
PRIMARY KEY(Id_number) );


5. GIVES
CREATE TABLE `GIVES`(
`CCemail` VARCHAR(30),
`Nid_number` VARCHAR(10),
PRIMARY KEY(CCemail,Nid_number),
FOREIGN KEY(CCemail) REFERENCES CHARITY_CENTER(email),
FOREIGN KEY(Nid_number) REFERENCES NEEDY(Id_number));


6. CAR_INSURANCE/HAS
DROP TABLE IF EXISTS `CAR_INSURANCE`;
CREATE TABLE `CAR_INSURANCE`(
`CEemployeeId` VARCHAR(10) NOT NULL,
`car_number` VARCHAR(15),
PRIMARY KEY(CEemployeeId,car_number),
FOREIGN KEY(CEemployeeId) REFERENCES COMPANY_EMPLOYEE(employeeId));
